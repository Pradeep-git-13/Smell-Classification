\section{Complete Algorithm & Equation Reference}

\subsection{Box-Muller Transform (Gaussian Generation)}

\textbf{Equation:}
$$Z = \mu + \sigma \sqrt{-2 \ln(U_1)} \cos(2\pi U_2)$$

Input: $U_1, U_2 \sim \text{Uniform}(0,1)$ independent

Output: $Z \sim \mathcal{N}(\mu, \sigma^2)$

Feature-specific $\sigma$ values:
\begin{itemize}
    \item temp: $\sigma = 3$
    \item hum: $\sigma = 6$
    \item mq2: $\sigma = 8$
    \item mq4: $\sigma = 6$
    \item mq8: $\sigma = 7$
    \item mq135: $\sigma = 9$
    \item mq136: $\sigma = 5$
    \item nh3: $\sigma = 0.3$
    \item vnh3: $\sigma = 0.3$
\end{itemize}

Value clamping after generation:
$$\text{clamped\_value} = \max(\text{range\_min}, \min(\text{range\_max}, Z))$$

\subsection{Class Statistics Computation}

Mean calculation:
$$\mu_f^{(c)} = \frac{1}{N_c} \sum_{i=1}^{N_c} x_f^{(i,c)}$$
where $N_c = 80$

Standard deviation:
$$\sigma_f^{(c)} = \sqrt{\frac{1}{N_c} \sum_{i=1}^{N_c} (x_f^{(i,c)} - \mu_f^{(c)})^2}$$

Safeguard: $\sigma_f^{(c)} \geq 1$ (prevent division by zero)

\subsection{Weighted Mahalanobis Distance}

Core distance formula:
$$D_c(\mathbf{x}) = \sum_{f=1}^{9} w_f \left(\frac{x_f - \mu_f^{(c)}}{\sigma_f^{(c)}}\right)^2$$

Normalized z-score per feature:
$$z_f = \frac{x_f - \mu_f^{(c)}}{\sigma_f^{(c)}}$$

Weight vector $\mathbf{w}$:
$$\mathbf{w} = [w_{\text{mq135}}, w_{\text{mq2}}, w_{\text{nh3}}, w_{\text{mq8}}, w_{\text{mq136}}, w_{\text{mq4}}, w_{\text{hum}}, w_{\text{temp}}, w_{\text{vnh3}}]$$
$$= [0.22, 0.18, 0.16, 0.12, 0.10, 0.09, 0.05, 0.05, 0.03]$$

\subsection{Softmax Probability (Sensor Classification)}

Score negation (distance to similarity):
$$\text{score}_c = -D_c(\mathbf{x})$$

Numerical stability (max subtraction):
$$s_c = \text{score}_c - \max\{\text{score}_1, \ldots, \text{score}_{12}\}$$

Exponential scaling (temperature $T = 3$):
$$e_c = \exp(3 \cdot s_c)$$

Softmax normalization:
$$P(c|\mathbf{x}) = \frac{e_c}{\sum_{c'=1}^{12} e_{c'}}$$

Final prediction:
$$\hat{c} = \arg\max_c P(c|\mathbf{x})$$

\subsection{Porter Stemming Algorithm}

Suffix removal rules (ordered):

\begin{enumerate}
    \item Check: word ends with "-ational"? Replace with "-ate"
    \item Check: word ends with "-tional"? Replace with "-tion"
    \item ... (18 more rules)
\end{enumerate}

Condition: Only apply if remaining word length $> 3$

Example: "roasting" $\to$ "roast" (remove "-ing")

\subsection{Tokenization Pipeline}

Step 1: Lowercase conversion
$$\text{text} \to \text{lowercase}(\text{text})$$

Step 2: Remove punctuation
$$\text{text} \to \text{regex}(/[^\textasciicircum a-z\textbackslash s]/g, " ")$$

Step 3: Split on whitespace
$$\text{tokens} = \text{split}(\text{text}, /\textbackslash s+/)$$

Step 4: Filter empty strings
$$\text{tokens} = \{\, t \in \text{tokens} : |t| > 1 \,\}$$

\subsection{Stop Word Filtering}

Define: $\text{isStop}(t) = t \in \text{STOP\_WORDS}$

Filter:
$$\text{meaningful} = \{\, t \in \text{tokens} : \neg \text{isStop}(t) \,\}$$

Result: Only content words remain

\subsection{Stemming Mapping}

Input: Each meaningful token $t$

Process:
\begin{enumerate}
    \item $t_{\text{stem}} = \text{stem}(t)$
    \item Store: $\{$original: $t$, stemmed: $t_{\text{stem}}$, isStop: False$\}$
\end{enumerate}

\subsection{Lexicon Matching Algorithm}

For each class $c$ and each token $t_{\text{stem}}$:

Check three conditions:
\begin{enumerate}
    \item $t_{\text{stem}} \equiv \text{keyword}_{\text{stem}}$ (exact match)
    \item $t_{\text{stem}} \subset \text{keyword}_{\text{stem}}$ (prefix match)
    \item $\text{keyword}_{\text{stem}} \subset t_{\text{stem}}$ (suffix match)
\end{enumerate}

If any match: $\text{matchedKW}_c \gets \text{matchedKW}_c \cup \{\text{keyword}\}$

\subsection{TF-IDF Scoring}

Term frequency:
$$\text{TF}(t) = \text{count}(t \text{ in document}) = |\{\, i : \text{token}_i = t \,\}|$$

Inverse document frequency:
$$\text{IDF}(t) = \log\left(\frac{\text{num\_classes}}{\text{num\_classes\_containing\_term}} + 1\right)$$

Combined score:
$$\text{TF-IDF}(t) = \text{TF}(t) \times \text{IDF}(t)$$

Class score aggregation:
$$\text{score}_c = \sum_{t \in \text{matchedKW}_c} \text{TF-IDF}(t)$$

Numeric example: If "ammonia" appears 1× (TF=1) and exists in 3 classes:
$$\text{TF-IDF}(\text{ammonia}) = 1 \times \log(12/3 + 1) = \log(5) \approx 1.609$$

\subsection{Softmax Probability (NLP Classification)}

Same softmax formula as sensor, but with temperature $T = 2$:

$$P(c|\text{text}) = \frac{\exp(2 \cdot s_c)}{\sum_{c'=1}^{12} \exp(2 \cdot s_{c'})}$$

\section{Complete Algorithm Execution Flows}

\subsection{Sensor Classification Flow}

\textbf{Input:} User provides 9 sensor values

\textbf{Step 1: Input Collection}
\begin{verbatim}
inputs = {
  temp: parseFloat(temp_val.value),
  hum: parseFloat(hum_val.value),
  mq2: parseFloat(mq2_val.value),
  mq4: parseFloat(mq4_val.value),
  mq8: parseFloat(mq8_val.value),
  mq135: parseFloat(mq135_val.value),
  mq136: parseFloat(mq136_val.value),
  nh3: parseFloat(nh3_val.value),
  vnh3: parseFloat(vnh3_val.value)
}
\end{verbatim}

\textbf{Step 2: Distance Calculation for Each Class}

For each of 12 classes:
\begin{enumerate}
    \item Retrieve CLASS\_STATS[classId].means
    \item Retrieve CLASS\_STATS[classId].stds
    \item Compute sum: $\sum_{f=1}^{9} w_f \left(\frac{x_f - \mu_f}{\sigma_f}\right)^2$
    \item Store as: scores[classId] = -(distance)
\end{enumerate}

\textbf{Step 3: Softmax Normalization}

Find max score: $s_{\max} = \max(\text{scores})$

For each class:
\begin{enumerate}
    \item $s_c \gets \text{scores}[c] - s_{\max}$
    \item $e_c \gets \exp(3 \times s_c)$
    \item Sum exponentials: $\text{sumExp} \gets \sum e_c$
\end{enumerate}

\textbf{Step 4: Probability Calculation}

For each class:
$$P(c|\mathbf{x}) = \frac{e_c}{\text{sumExp}}$$

\textbf{Step 5: Prediction \& Confidence}

$\hat{c} = \arg\max_c P(c|\mathbf{x})$

confidence $= P(\hat{c}) \times 100\%$

\textbf{Step 6: Top-6 Sorting}

Sort (classId, probability) pairs descending

Take first 6 for display

\textbf{Step 7: Feature Importance}

Display FEATURE\_WEIGHTS in descending order

\textbf{Step 8: Result Rendering}

\begin{enumerate}
    \item Set result-icon: SMELL\_CLASSES[\^c].icon
    \item Set result-name: SMELL\_CLASSES[\^c].name
    \item Set result-conf: formatted confidence string
    \item Populate conf-bars with top-6
    \item Populate metadata fields
    \item Populate feat-bars
    \item Scroll to result section
\end{enumerate}

\textbf{Output:} Full results UI with icon, name, confidence, bars

\subsection{NLP Classification Flow}

\textbf{Input:} User types natural language description

\textbf{Step 1: Tokenization}

"It smells sharp and pungent, like ammonia!"

\begin{verbatim}
lowercase: "it smells sharp and pungent, like ammonia!"
remove punctuation: "it smells sharp and pungent  like ammonia "
split: ["it", "smells", "sharp", "and", "pungent", "like", "ammonia"]
\end{verbatim}

\textbf{Step 2: Stop Word Filtering}

Marked as stop: \{it, and, like\}

Meaningful: \{smells, sharp, pungent, ammonia\}

\textbf{Step 3: Stemming}

\begin{verbatim}
smells -> smell (remove -s)
sharp -> sharp (no suffix)
pungent -> pungent (no suffix)
ammonia -> ammonia (no suffix)
\end{verbatim}

\textbf{Step 4: Class Scoring}

For each class, check if stemmed tokens match lexicon keywords:

Example for ammonia class:
\begin{enumerate}
    \item "pungent" matches SMELL\_LEXICON["ammonia"] (direct hit)
    \item "ammonia" matches SMELL\_LEXICON["ammonia"] (direct hit)
    \item "sharp" matches SMELL\_LEXICON["ammonia"] (direct hit)
\end{enumerate}

TF-IDF calculation:
\begin{enumerate}
    \item TF(pungent) = 1
    \item IDF(pungent) = $\log(12/3 + 1) = \log(5) \approx 1.609$
    \item TF-IDF(pungent) $\approx 1.609$
    \item Similarly for "ammonia" and "sharp"
\end{enumerate}

Total score for ammonia: sum of TF-IDF values

\textbf{Step 5: Softmax with T=2}

Numerator: $\exp(2 \times s_c)$

Normalize across all 12 classes

\textbf{Step 6: Prediction}

$\hat{c}_{\text{nlp}} = \arg\max_c P(c|\text{text})$

\textbf{Step 7: Token Visualization}

Render tokens as pills:
\begin{enumerate}
    \item Stop words: gray, strikethrough
    \item Matched keywords: green/accent
    \item Unmatched: gray
\end{enumerate}

\textbf{Step 8: Result Display (same as sensor path)}

\textbf{Output:} Full results UI with matched keywords highlighted

\subsection{Data Generation Flow (Initialization)}

\textbf{For each class c in SMELL\_CLASSES:}

\begin{enumerate}
    \item Retrieve SMELL\_PROFILES[c.id]
    \item For iteration i = 1 to 80:
    \begin{enumerate}
        \item For each feature f:
        \begin{enumerate}
            \item Generate $U_1, U_2 \sim \text{Uniform}(0,1)$
            \item Apply Box-Muller:
            $$Z = \text{profile}[f] + \sigma_f \sqrt{-2\ln U_1} \cos(2\pi U_2)$$
            \item Clamp to valid range
            \item Store in sample
        \end{enumerate}
        \item Add sample to DATASET
    \end{enumerate}
\end{enumerate}

Total: 960 samples created

\textbf{Shuffle DATASET}

Fisher-Yates shuffle (random ordering)

\textbf{Compute CLASS\_STATS}

For each class, compute means and stds from 80 samples

\textbf{Output:} Ready-to-use DATASET and CLASS\_STATS
